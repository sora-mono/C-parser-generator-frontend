43 bool DfaGenerator::DfaConstruct() {
44  nfa_generator_.MergeOptimization();
45  NfaNodeId nfa_head_handler = nfa_generator_.GetHeadNfaNodeId();
46  auto&& [set_now, tail_data] = nfa_generator_.Closure(nfa_head_handler);
47  assert(tail_data == NfaGenerator::kNotTailNodeTag);
48  auto [root_itermediate_node_id, inserted] = InOrInsert(std::move(set_now));
49  assert(inserted);
50  std::queue<IntermediateNodeId> q;
51  q.push(root_itermediate_node_id);
52  root_intermediate_node_id_ = root_itermediate_node_id;
53  while (!q.empty()) {
54  IntermediateNodeId intermediate_node_handler_now = q.front();
55  q.pop();
56  SetId set_handler_now =
57  GetIntermediateNode(intermediate_node_handler_now).set_handler;
58  for (int i = CHAR_MIN; i <= CHAR_MAX; i++) {
59  auto [transformed_intermediate_node_id, inserted] =
60  SetGoto(set_handler_now, static_cast<char>(i));
61  if (!transformed_intermediate_node_id.IsValid()) {
62  // 该字符下不可转移
63  continue;
64  }
65  // 设置转移条件
66  GetIntermediateNode(intermediate_node_handler_now)
67  .forward_nodes[static_cast<char>(i)] =
68  transformed_intermediate_node_id;
69  if (inserted) {
70  // 新集合对应的中间节点以前不存在，插入队列等待处理
71  q.push(transformed_intermediate_node_id);
72  }
73  }
74  }
75  DfaMinimize();
76  return true;
77 }

79 bool DfaGenerator::DfaMinimize() {
80  transform_array_size_ = 0; // 清零最终有效节点数
81  std::list<IntermediateNodeId> nodes;
82  for (auto iter = node_manager_intermediate_node_.Begin();
83  iter != node_manager_intermediate_node_.End(); ++iter) {
84  nodes.push_back(iter.GetId());
85  }
86  IntermediateNodeClassify(std::move(nodes));
87  dfa_config_.resize(transform_array_size_);
88  for (auto& p : dfa_config_) {
89  // 填充整个转移表为无法转移状态
90  p.first.fill(TransformArrayId::InvalidId());
91  }
92  // 标记转移表条目是否完成构建
93  std::vector<bool> logged_index(transform_array_size_, false);
94  for (auto& p : intermediate_node_to_final_node_) {
95  TransformArrayId index = p.second;
96  IntermediateDfaNode& intermediate_node = GetIntermediateNode(p.first);
97  if (logged_index[index] == false) {
98  // 该状态转移表条目未配置
99  dfa_config_[index].second = intermediate_node.word_attached_data;
100  for (int i = CHAR_MIN; i <= CHAR_MAX; i++) {
101  // 寻找有效节点并设置语法分析表中对应项
102  // 从最小的char开始
103  IntermediateNodeId next_node_id =
104  intermediate_node.forward_nodes[static_cast<char>(i)];
105  if (next_node_id.IsValid()) [[unlikely]] {
106  // 该条件下可以转移，查询转移到的中间节点对应转移表条目ID
107  auto iter = intermediate_node_to_final_node_.find(next_node_id);
108  assert(iter != intermediate_node_to_final_node_.end());
109  dfa_config_[index].first[static_cast<char>(i)] = iter->second;
110  }
111  }
112  logged_index[index] = true;
113  }
114  }
115 #ifdef _DEBUG
116  // 检查每个转移表条目都被填写过
117  for (bool logged : logged_index) {
118  assert(logged);
119  }
120 #endif // _DEBUG
121  root_transform_array_id_ =
122  intermediate_node_to_final_node_.find(root_intermediate_node_id_)->second;
123  assert(root_transform_array_id_.IsValid());
124  root_intermediate_node_id_ = IntermediateNodeId::InvalidId();
125  node_manager_intermediate_node_.ObjectManagerInit();
126  node_manager_intermediate_node_.ShrinkToFit();
127  intermediate_node_to_final_node_.clear();
128  return true;
129 }

131 std::pair<DfaGenerator::IntermediateNodeId, bool> DfaGenerator::SetGoto(
132  SetId set_src, char c_transform) {
133  SetType set;
134  TailNodeData tail_data(NfaGenerator::kNotTailNodeTag);
135  for (auto x : GetSetObject(set_src)) {
136  auto [set_temp, tail_data_temp] = nfa_generator_.Goto(x, c_transform);
137  if (set_temp.empty()) [[likely]] {
138  continue;
139  }
140  set.merge(std::move(set_temp));
141  // 不存在尾节点标记或新的标记优先级大于原来的标记则修改
142  if (tail_data_temp != NfaGenerator::kNotTailNodeTag) [[unlikely]] {
143  if (tail_data == NfaGenerator::kNotTailNodeTag ||
144  tail_data_temp.second > tail_data.second) {
145  tail_data = tail_data_temp;
146  }
147  }
148  }
149  if (set.empty()) {
150  return std::make_pair(IntermediateNodeId::InvalidId(), false);
151  } else {
152  return InOrInsert(std::move(set), std::move(tail_data.first));
153  }
154 }

169 void DfaGenerator::IntermediateNodeClassify(
170  std::list<IntermediateNodeId>&& node_ids, char c_transform) {
171  // 存储不同分组，键值为当前转移条件下转移到的节点句柄
172  // 无法转移的节点键值前半部分使用IntermediateNodeId::InvalidId()
173  std::unordered_map<std::pair<IntermediateNodeId, WordAttachedData>,
174  std::list<IntermediateNodeId>,
175  PairOfIntermediateNodeIdAndWordAttachedDataHasher>
176  transform_node_ids;
177  for (auto handler : node_ids) {
178  IntermediateNodeId next_node_id = IntermediateGoto(handler, c_transform);
179  IntermediateDfaNode& node = GetIntermediateNode(handler);
180  transform_node_ids[std::make_pair(next_node_id, node.word_attached_data)]
181  .push_back(handler);
182  }
183  if (c_transform == CHAR_MAX) [[unlikely]] {
184  // 最后一个转移条件已经判断，应填写中间节点到转移表条目的映射表
185  for (auto& group : transform_node_ids) {
186  for (auto& intermediate_node_id : group.second) {
187  // 当前具有的转移表条目数为新建条目的下标
188  intermediate_node_to_final_node_.emplace(intermediate_node_id,
189  transform_array_size_);
190  }
191  ++transform_array_size_;
192  }
193  } else {
194  // 未达到最后条件（CHAR_MAX），应继续对数目大于1的组分类
195  // 修改转移条件
196  ++c_transform;
197  for (auto& group : transform_node_ids) {
198  if (group.second.size() > 1) {
199  // 组内成员大于1个，需要继续分类
200  IntermediateNodeClassify(std::move(group.second), c_transform);
201  } else {
202  // 组内只有一个成员，直接填写中间节点到转移表条目的映射表
203  intermediate_node_to_final_node_.emplace(group.second.front(),
204  transform_array_size_);
205  ++transform_array_size_;
206  }
207  }
208  }
209 }
