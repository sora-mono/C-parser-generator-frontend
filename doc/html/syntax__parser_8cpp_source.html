35 void SyntaxParser::TerminalWordWaitingProcess() {
36  assert(GetWaitingProcessWordInfo().word_attached_data_.node_type ==
37  ProductionNodeType::kTerminalNode ||
38  GetWaitingProcessWordInfo().word_attached_data_.node_type ==
39  ProductionNodeType::kOperatorNode ||
40  GetWaitingProcessWordInfo().word_attached_data_.node_type ==
41  ProductionNodeType::kEndNode);
42  ProductionNodeId production_node_to_shift_id =
43  GetWaitingProcessWordInfo().word_attached_data_.production_node_id;
44  const ActionAndAttachedDataInterface& action_and_attached_data =
45  *GetActionAndTarget(GetParsingDataNow().syntax_analysis_table_entry_id,
46  production_node_to_shift_id);
47  switch (action_and_attached_data.GetActionType()) {
48  // TODO 添加接受时的后续处理
49  [[unlikely]] case ActionType::kAccept : exit(0);
50  break;
51  // TODO 添加错误处理功能
52  [[unlikely]] case ActionType::kError : std::cerr << std::format("语法错误")
53  << std::endl;
54  exit(-1);
55  break;
56  case ActionType::kReduct:
57  Reduct(action_and_attached_data.GetReductAttachedData());
58  break;
59  case ActionType::kShift:
60  ShiftTerminalWord(action_and_attached_data.GetShiftAttachedData());
61  break;
62  case ActionType::kShiftReduct: {
63  // 需要根据实际情况判断移入还是规约
64  auto& terminal_node_info =
65  GetWaitingProcessWordInfo().word_attached_data_;
66  switch (terminal_node_info.node_type) {
67  case ProductionNodeType::kTerminalNode:
68  // 非运算符节点使用贪心策略，防止有公共前缀的产生式只有最短的那条有效
69  ShiftTerminalWord(action_and_attached_data.GetShiftAttachedData());
70  break;
71  case ProductionNodeType::kOperatorNode: {
72  // 运算符优先级必须不为0
73  OperatorPriority priority_now = GetParsingDataNow().operator_priority;
74  auto [operator_associate_type, operator_priority] =
75  terminal_node_info.GetAssociatityTypeAndPriority(
76  LastOperateIsReduct());
77  if (priority_now > operator_priority) {
78  // 当前优先级高于待处理的运算符的优先级，执行规约操作
79  Reduct(action_and_attached_data.GetReductAttachedData());
80  } else if (priority_now == operator_priority) {
81  // 当前优先级等于待处理的运算符的优先级，需要判定结合性
82  if (operator_associate_type ==
83  OperatorAssociatityType::kLeftToRight) {
84  // 运算符为从左到右结合，执行规约操作
85  Reduct(action_and_attached_data.GetReductAttachedData());
86  } else {
87  // 运算符为从右到左结合，执行移入操作
88  ShiftTerminalWord(
89  action_and_attached_data.GetShiftAttachedData());
90  }
91  } else {
92  // 当前优先级低于待处理的运算符的优先级，执行移入操作
93  ShiftTerminalWord(action_and_attached_data.GetShiftAttachedData());
94  }
95  } break;
96  case ProductionNodeType::kEndNode:
97  // EndNode在转移表里的动作只能规约不能移入
98  case ProductionNodeType::kNonTerminalNode:
99  // NonTerminalNode在产生后就被Reduct函数调用ShiftNonTerminalNode消耗
100  default:
101  assert(false);
102  break;
103  }
104  } break;
105  default:
106  assert(false);
107  break;
108  }
109 }

111 inline void SyntaxParser::ShiftTerminalWord(
112  const ShiftAttachedData& action_and_target) {
113  ParsingData& parsing_data_now = GetParsingDataNow();
114  WordInfo& word_info = GetWaitingProcessWordInfo();
115  // 构建当前单词的数据
116  // 待移入节点的ID
117  parsing_data_now.shift_node_id =
118  word_info.word_attached_data_.production_node_id;
119  // 添加待移入节点信息到当前解析用信息
120  parsing_data_now.word_data_to_user.SetWordDataToUser(
121  TerminalWordData{.word = std::move(word_info.symbol_)});
122  // 如果移入了运算符则更新优先级为新的优先级
123  OperatorPriority new_parsing_data_priority;
124  if (word_info.word_attached_data_.node_type ==
125  ProductionNodeType::kOperatorNode) {
126  new_parsing_data_priority =
127  word_info.word_attached_data_
128  .GetAssociatityTypeAndPriority(LastOperateIsReduct())
129  .second;
130  } else {
131  // 否则使用原来的优先级
132  new_parsing_data_priority = parsing_data_now.operator_priority;
133  }
134  // 压入移入该单词后得到的解析数据
135  PushParsingData(
136  ParsingData{.syntax_analysis_table_entry_id =
137  action_and_target.GetNextSyntaxAnalysisTableEntryId(),
138  .operator_priority = new_parsing_data_priority});
139  // 获取下一个单词
140  GetNextWord();
141  // 执行了移入操作，需要设置上一步为非规约操作
142  SetLastOperateIsNotReduct();
143 }

145 void SyntaxParser::Reduct(const ReductAttachedData& action_and_target) {
146  const ReductAttachedData& reduct_attached_data =
147  action_and_target.GetReductAttachedData();
148  const ProcessFunctionInterface& process_function_object =
149  GetProcessFunctionClass(reduct_attached_data.GetProcessFunctionClassId());
150  const auto& production_body = reduct_attached_data.GetProductionBody();
151  // 传递给用户定义函数的数据
152  std::vector<WordDataToUser> word_data_to_user;
153  // 预分配空间，防止多次扩容
154  word_data_to_user.resize(production_body.size());
155 
156  // 弹出栈顶数据，露出word_data_to_user存储有效数据的部分
157  // 除了这里任何时候栈顶数据的word_data_to_user都不包含有效数据
158  ParsingData old_parsing_data = std::move(GetParsingDataNow());
159  PopTopParsingData();
160  // 生成规约数据，数组内数据顺序为产生式书写顺序
161  // 从后向前添加数据
162  auto production_node_id_iter = production_body.rbegin();
163  for (auto user_data_iter = word_data_to_user.rbegin();
164  user_data_iter != word_data_to_user.rend(); ++production_node_id_iter) {
165  if (GetParsingDataNow().shift_node_id == *production_node_id_iter)
166  [[likely]] {
167  // 当前产生式正常参与规约，获取之前保存下来提供给用户规约使用的数据
168  *user_data_iter = std::move(GetParsingDataNow().word_data_to_user);
169  ++user_data_iter;
170  // 弹出解析数据栈顶部无附属数据的部分，露出之前存入的解析数据
171  old_parsing_data = std::move(GetParsingDataNow());
172  PopTopParsingData();
173  // 由于有哨兵，解析数据栈永远不空
174  assert(GetParsingStackSize() != 0);
175  } else {
176  // 该节点空规约，设置空规约节点标记
177  // 空规约节点存储std::monostate
178  user_data_iter->SetWordDataToUser(std::monostate{});
179  ++user_data_iter;
180  }
181  }
182  // 恢复栈顶数据
183  // 栈顶数据储存有效信息：当前语法分析表条目和运算符优先级
184  PushParsingData(std::move(old_parsing_data));
185  ShiftNonTerminalWord(
186  process_function_object.Reduct(std::move(word_data_to_user)),
187  action_and_target.GetReductAttachedData().GetReductedNonTerminalNodeId());
188  // 执行了一次完整的规约操作，需要设置上一步执行了规约操作的标记
189  SetLastOperateIsReduct();
190 }
