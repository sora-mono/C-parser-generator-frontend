41 NfaGenerator::Closure(NfaNodeId node_id) {
42  std::unordered_set<NfaNodeId> result_set;
43  TailNodeData word_attached_data(kNotTailNodeTag);
44  std::queue<NfaNodeId> q;
45  for (auto x : node_manager_.GetIdsReferringSameObject(node_id)) {
46  q.push(x);
47  }
48  while (!q.empty()) {
49  NfaNodeId id_now = q.front();
50  q.pop();
51  if (result_set.find(id_now) != result_set.end()) {
52  continue;
53  }
54  auto [result_iter, inserted] = result_set.insert(id_now);
55  assert(inserted);
56  auto iter = tail_nodes_.find(id_now);
57  // 判断是否为尾节点
58  if (iter != tail_nodes_.end()) {
59  TailNodeData& tail_node_data_new = iter->second;
60  WordPriority priority_old = word_attached_data.second;
61  WordPriority priority_new = tail_node_data_new.second;
62  if (word_attached_data == kNotTailNodeTag) {
63  // 以前无尾节点记录
64  word_attached_data = tail_node_data_new;
65  } else if (priority_new > priority_old) {
66  // 当前记录优先级大于以前的优先级
67  word_attached_data = tail_node_data_new;
68  } else if (priority_new == priority_old &&
69  tail_node_data_new.first != word_attached_data.first) {
70  // 两个尾节点标记优先级相同，对应尾节点不同
71  // 输出错误信息
72  std::cerr
73  << std::format(
74  "NfaGenerator "
75  "Error:"
76  "存在两个正则表达式在相同的输入下均可获取单词，且单词优先级"
77  "相同，产生歧义，请检查终结节点定义/运算符定义部分")
78  << std::endl;
79  assert(false);
80  exit(-1);
81  }
82  }
83  for (auto x : GetNfaNode(id_now).GetUnconditionTransferNodesIds()) {
84  q.push(x);
85  }
86  }
87  return std::make_pair(std::move(result_set), std::move(word_attached_data));
88 }

139 std::pair<NfaGenerator::NfaNodeId, NfaGenerator::NfaNodeId>
140 NfaGenerator::RegexConstruct(TailNodeData&& tail_node_data,
141  const std::string& raw_regex_string,
142  size_t&& next_character_index,
143  const bool add_to_nfa_head) {
144  NfaNodeId head_id = node_manager_.EmplaceObject();
145  NfaNodeId tail_id = head_id;
146  NfaNodeId pre_tail_id = head_id;
147  if (next_character_index >= raw_regex_string.size()) [[unlikely]] {
148  return std::make_pair(head_id, tail_id);
149  }
150  char c_now;
151  while (next_character_index < raw_regex_string.size()) {
152  c_now = raw_regex_string[next_character_index];
153  ++next_character_index;
154  switch (c_now) {
155  case '[': {
156  auto [temp_head_id, temp_tail_id] =
157  CreateSwitchTree(raw_regex_string, &next_character_index);
158  if (!(temp_head_id.IsValid() && temp_tail_id.IsValid())) [[unlikely]] {
159  std::cerr << std::format("非法正则 {:}\n{: >{}}", raw_regex_string,
160  '^', next_character_index + 9)
161  << std::endl;
162  exit(-1);
163  }
164  GetNfaNode(tail_id).AddNoconditionTransfer(temp_head_id);
165  pre_tail_id = tail_id;
166  tail_id = temp_tail_id;
167  } break;
168  case ']':
169  // regex_construct函数不应该处理]字符，应交给create_switch_tree处理
170  assert(false);
171  break;
172  case '(': {
173  auto [temp_head_id, temp_tail_id] =
174  RegexConstruct(std::move(tail_node_data), raw_regex_string,
175  std::move(next_character_index), false);
176  if (!(temp_head_id.IsValid() && temp_tail_id.IsValid())) [[unlikely]] {
177  std::cerr << std::format("非法正则 {:}\n{: >{}}", raw_regex_string,
178  '^', next_character_index + 9)
179  << std::endl;
180  exit(-1);
181  }
182  GetNfaNode(tail_id).AddNoconditionTransfer(temp_head_id);
183  pre_tail_id = tail_id;
184  tail_id = temp_tail_id;
185  } break;
186  case ')':
187  return std::make_pair(head_id, tail_id);
188  break;
189  case '+': // 对单个字符和组合结构均生效
190  GetNfaNode(tail_id).AddNoconditionTransfer(pre_tail_id);
191  break;
192  case '*': // 对单个字符和组合结构均生效
193  GetNfaNode(tail_id).AddNoconditionTransfer(pre_tail_id);
194  GetNfaNode(pre_tail_id).AddNoconditionTransfer(tail_id);
195  break;
196  case '?': // 对单个字符和组合结构均生效
197  GetNfaNode(pre_tail_id).AddNoconditionTransfer(tail_id);
198  break;
199  case '\\': // 仅对单个字符生效
200  if (next_character_index >= raw_regex_string.size()) [[unlikely]] {
201  std::cerr << std::format("非法正则 {:}\n{: >{}}", raw_regex_string,
202  '^', next_character_index + 9)
203  << std::endl;
204  exit(-1);
205  }
206  c_now = raw_regex_string[next_character_index];
207  ++next_character_index;
208  pre_tail_id = tail_id;
209  tail_id = node_manager_.EmplaceObject();
210  GetNfaNode(pre_tail_id).SetConditionTransfer(c_now, tail_id);
211  break;
212  case '.': // 仅对单个字符生效
213  pre_tail_id = tail_id;
214  tail_id = node_manager_.EmplaceObject();
215  for (char transform_char = CHAR_MIN; transform_char != CHAR_MAX;
216  ++transform_char) {
217  GetNfaNode(pre_tail_id).SetConditionTransfer(transform_char, tail_id);
218  }
219  GetNfaNode(pre_tail_id).SetConditionTransfer(CHAR_MAX, tail_id);
220  break;
221  default:
222  pre_tail_id = tail_id;
223  tail_id = node_manager_.EmplaceObject();
224  GetNfaNode(pre_tail_id).SetConditionTransfer(c_now, tail_id);
225  break;
226  }
227  }
228  if (head_id != tail_id) [[likely]] {
229  if (add_to_nfa_head) {
230  GetNfaNode(head_node_id_).AddNoconditionTransfer(head_id);
231  SetTailNode(tail_id, std::move(tail_node_data));
232  }
233  } else {
234  node_manager_.RemoveObject(head_id);
235  head_id = tail_id = NfaNodeId::InvalidId();
236  }
237  return std::make_pair(head_id, tail_id);
238 }

256 void NfaGenerator::MergeOptimization() {
257  node_manager_.SetAllObjectsCanBeSourceInMerge();
258  std::queue<NfaNodeId> q;
259  q.push(GetHeadNfaNodeId());
260  while (!q.empty()) {
261  NfaNodeId id_now = q.front();
262  NfaNode& node_now = GetNfaNode(id_now);
263  q.pop();
264  if (!node_manager_.CanBeSourceInMerge(id_now)) {
265  continue;
266  }
267  // 检查每一个与当前处理节点等价的节点（就是当前节点可以无条件转移到的节点）
268  // 如果当前节点转移表中的项在等价节点中存在则可以从当前节点转移表中删除
269  for (auto equal_node_id :
270  GetNfaNode(id_now).GetUnconditionTransferNodesIds()) {
271  if (equal_node_id == id_now) [[unlikely]] {
272  // 跳过转移到自己的情况
273  continue;
274  }
275  auto& equal_node = GetNfaNode(equal_node_id);
276  // 处理无条件转移表
277  const auto& equal_node_unconditional_transfer_node_ids =
278  equal_node.GetUnconditionTransferNodesIds();
279  for (auto iter = node_now.GetUnconditionTransferNodesIds().cbegin();
280  iter != node_now.GetUnconditionTransferNodesIds().cend();) {
281  if (equal_node_unconditional_transfer_node_ids.find(*iter) !=
282  equal_node_unconditional_transfer_node_ids.end()) {
283  // 当前节点转移表中的项在等价节点中存在
284  // 移除该项
285  if (*iter != equal_node_id) [[likely]] {
286  // 如果存在等价节点的无条件自环节点则不能移除
287  // 否则会失去指向等价节点的记录，导致正则不完整、内存泄漏等
288  ++iter;
289  node_now.RemoveConditionlessTransfer(*iter);
290  // continue防止额外前移iter
291  continue;
292  }
293  }
294  ++iter;
295  }
296  // 处理条件转移表
297  const auto& equal_node_conditional_transfers =
298  equal_node.GetConditionalTransfers();
299  for (auto iter = node_now.GetConditionalTransfers().cbegin();
300  iter != node_now.GetConditionalTransfers().cend();) {
301  auto equal_node_iter =
302  equal_node_conditional_transfers.find(iter->first);
303  if (equal_node_iter != equal_node_conditional_transfers.cend() &&
304  iter->second == equal_node_iter->second) {
305  // 当前节点转移表中的项在等价节点中存在
306  // 移除该项
307  ++iter;
308  node_now.RemoveConditionalTransfer(iter->first);
309  // continue防止额外前移iter
310  continue;
311  }
312  ++iter;
313  }
314  // 压入等效节点等待处理
315  q.push(equal_node_id);
316  // 压入当前节点的所有可以条件转移到的节点等待处理
317  for (const auto& conditional_transfer :
318  node_now.GetConditionalTransfers()) {
319  q.push(conditional_transfer.second);
320  }
321  }
322  // 检查是否可以将node_now与node_now唯一可无条件转移到的节点合并
323  if (node_now.GetConditionalTransfers().empty() &&
324  node_now.GetUnconditionTransferNodesIds().size() == 1) [[unlikely]] {
325  // 只剩一条无条件转移路径，该节点可以与无条件转移到的节点合并
326  NfaNodeId dst_node_id =
327  *node_now.GetUnconditionTransferNodesIds().cbegin();
328  const NfaGenerator::TailNodeData& dst_tag = GetTailNodeData(dst_node_id);
329  const NfaGenerator::TailNodeData& src_tag = GetTailNodeData(id_now);
330  if (dst_tag != NfaGenerator::kNotTailNodeTag &&
331  src_tag != NfaGenerator::kNotTailNodeTag &&
332  dst_tag.second == src_tag.second && dst_tag.first != src_tag.first) {
333  std::cerr << std::format("两个尾节点具有相同优先级且不对应同一个节点")
334  << std::endl;
335  exit(-1);
336  }
337  bool result = GetNfaNode(dst_node_id).MergeNodes(&GetNfaNode(id_now));
338  if (result) {
339  if (src_tag != NfaGenerator::kNotTailNodeTag) {
340  SetTailNode(dst_node_id, src_tag);
341  RemoveTailNode(id_now);
342  }
343  bool result = node_manager_.MergeObjects(
344  dst_node_id, id_now,
345  [](NfaNode&, NfaNode&) -> bool { return true; });
346  assert(result);
347  }
348 
349  } else {
350  // 没有执行任何操作，不存在从该节点开始的合并操作
351  // 设置该节点在合并时不能作为源节点
352  node_manager_.SetObjectCanNotBeSourceInMerge(id_now);
353  }
354  }
355 }

359 std::pair<NfaGenerator::NfaNodeId, NfaGenerator::NfaNodeId>
360 NfaGenerator::CreateSwitchTree(const std::string& raw_regex_string,
361  size_t* const next_character_index) {
362  NfaNodeId head_id = node_manager_.EmplaceObject();
363  NfaNodeId tail_id = node_manager_.EmplaceObject();
364  NfaNode& head_node = GetNfaNode(head_id);
365  char character_pre;
366  // 初始化成不是']'的值从而可以进入循环
367  char character_now = '[';
368  while (character_now != ']') {
369  if (*next_character_index >= raw_regex_string.size()) [[unlikely]] {
370  std::cerr << std::format("非法正则 {:}\n{: >{}}", raw_regex_string, '^',
371  *next_character_index + 9)
372  << std::endl;
373  exit(-1);
374  }
375  character_pre = character_now;
376  character_now = raw_regex_string[*next_character_index];
377  ++*next_character_index;
378  switch (character_now) {
379  case '-': {
380  // 读入另一端的字符
381  if (*next_character_index >= raw_regex_string.size()) [[unlikely]] {
382  std::cerr << std::format("非法正则 {:}\n{: >{}}", raw_regex_string,
383  '^', *next_character_index + 9)
384  << std::endl;
385  exit(-1);
386  }
387  character_now = raw_regex_string[*next_character_index];
388  ++*next_character_index;
389  if (character_now == ']') [[unlikely]] {
390  // [+-]这种正则表达式
391  break;
392  }
393  char bigger_character = std::max(character_now, character_pre);
394  for (char smaller_character = std::min(character_pre, character_now);
395  smaller_character < bigger_character; smaller_character++) {
396  head_node.SetConditionTransfer(smaller_character, tail_id);
397  }
398  // 防止bigger_character == CHAR_MAX导致回环，无法退出上面循环
399  head_node.SetConditionTransfer(bigger_character, tail_id);
400  } break;
401  case '\\':
402  if (*next_character_index >= raw_regex_string.size()) [[unlikely]] {
403  std::cerr << std::format("非法正则 {:}\n{: >{}}", raw_regex_string,
404  '^', *next_character_index + 9)
405  << std::endl;
406  exit(-1);
407  }
408  character_now = raw_regex_string[*next_character_index];
409  ++*next_character_index;
410  head_node.SetConditionTransfer(character_now, tail_id);
411  break;
412  case ']':
413  break;
414  default:
415  head_node.SetConditionTransfer(character_now, tail_id);
416  break;
417  }
418  }
419  if (head_node.GetConditionalTransfers().empty()) [[unlikely]] {
420  node_manager_.RemoveObject(head_id);
421  std::cerr << std::format("非法正则 {:}\n{: >{}}\n[]中为空",
422  raw_regex_string, '^', *next_character_index + 9)
423  << std::endl;
424  exit(-1);
425  }
426  return std::make_pair(head_id, tail_id);
427 }
