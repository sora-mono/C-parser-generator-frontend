441 void SyntaxGenerator::GetNonTerminalNodeFirstNodeIds(
442  ProductionNodeId production_node_id, ForwardNodesContainer* result,
443  std::unordered_set<ProductionNodeId>&& processed_nodes) {
444  bool inserted = processed_nodes.insert(production_node_id).second;
445  assert(inserted);
446  assert(production_node_id.IsValid());
447  assert(result);
448  NonTerminalProductionNode& production_node =
449  static_cast<NonTerminalProductionNode&>(
450  GetProductionNode(production_node_id));
451  assert(production_node.GetType() == ProductionNodeType::kNonTerminalNode);
452  for (auto& body : production_node.GetAllBody()) {
453  NextWordToShiftIndex node_id_index(0);
454  // 正在处理的产生式节点的ID
455  ProductionNodeId node_id;
456  while (node_id_index < body.production_body.size()) {
457  node_id = body.production_body[node_id_index];
458  if (GetProductionNode(node_id).GetType() !=
459  ProductionNodeType::kNonTerminalNode) {
460  result->insert(node_id);
461  break;
462  }
463  assert(node_id.IsValid());
464  // 检查该节点是否已经处理过
465  if (processed_nodes.find(node_id) == processed_nodes.end()) {
466  // 如果该节点未处理则获取该节点的first节点ID并添加到主first节点ID集合中
467  GetNonTerminalNodeFirstNodeIds(node_id, result,
468  std::move(processed_nodes));
469  }
470  if (static_cast<NonTerminalProductionNode&>(GetProductionNode(node_id))
471  .CouldBeEmptyReduct()) {
472  // 该非终结节点可以空规约，需要考虑后面的节点
473  ++node_id_index;
474  } else {
475  // 已处理所有节点，跳出循环
476  break;
477  }
478  }
479  }

647 ProductionItemSetId SyntaxGenerator::GetProductionItemSetIdFromProductionItems(
648  const std::list<SyntaxGenerator::ProductionItemAndForwardNodesContainer::
649  const_iterator>& items) {
650 #ifdef _DEBUG
651  // 检查参数中不存在重复项
652  // 所有的参数都是移入相同的节点前的项，移入相同节点后的项相同的充分必要条件为
653  // 移入相同节点前的项相同
654  // 所以只需检查移入相同节点前的项是否相同
655  for (auto iter = items.cbegin(); iter != items.cend(); ++iter) {
656  auto compare_iter = iter;
657  while (++compare_iter != items.cend()) {
658  assert((*iter)->first != (*compare_iter)->first);
659  }
660  }
661 #endif // _DEBUG
662 
663  // 存储项集ID包含几个给定项的个数，只有包含全部给定项才可能成为这些项属于的项集
664  std::unordered_map<ProductionItemSetId, size_t>
665  production_item_set_includes_item_size;
666  // 统计项集包含给定项的个数
667  for (const auto& item : items) {
668  // 拆分移入节点前的项数据
669  auto [production_node_id, production_body_id, next_word_to_shift_index] =
670  item->first;
671  // 获取移入节点后的项数据
672  ++next_word_to_shift_index;
673  const auto& production_item_set_ids_item_belong_to =
674  GetProductionItemSetIdFromProductionItem(
675  production_node_id, production_body_id, next_word_to_shift_index);
676  // 优化手段，任意给定项不属于任何项集则不存在包含所有给定项的项集
677  if (production_item_set_ids_item_belong_to.empty()) [[likely]] {
678  return ProductionItemSetId::InvalidId();
679  }
680  for (auto production_item_set_id : production_item_set_ids_item_belong_to) {
681  auto [iter, inserted] = production_item_set_includes_item_size.emplace(
682  production_item_set_id, 1);
683  if (!inserted) {
684  // 该项集ID已经记录，增加该项集ID出现的次数
685  ++iter->second;
686  }
687  }
688  }
689  // 检查所有出现过的项集ID，找出包含所有给定项的项集ID
690  auto iter = production_item_set_includes_item_size.begin();
691  for (; iter != production_item_set_includes_item_size.end(); ++iter) {
692  if (iter->second != items.size()) [[likely]] {
693  // 该项集没包含所有项
694  continue;
695  }
696  // 检查是否所有项都是该项集的核心项
697  bool all_item_is_main_item = true;
698  ProductionItemSet& production_item_set_may_belong_to =
699  GetProductionItemSet(iter->first);
700  for (const auto& item : items) {
701  // 构造移入节点后得到的项
702  auto item_after_shift = item->first;
703  ++std::get<NextWordToShiftIndex>(item_after_shift);
704  if (!production_item_set_may_belong_to.IsMainItem(item_after_shift)) {
705  all_item_is_main_item = false;
706  break;
707  }
708  }
709  if (all_item_is_main_item) {
710  return iter->first;
711  }
712  }
713  return ProductionItemSetId::InvalidId();
714 }

754  next_word_to_shift_index);

881  std::list<SyntaxAnalysisTableEntryId>&& syntax_analysis_table_entry_ids,
882  std::vector<std::list<SyntaxAnalysisTableEntryId>>* equivalent_ids) {
883  if (index >= terminal_node_ids.size()) [[unlikely]] {
884  // 分类完成，将可以合并的组添加到结果

890  ProductionNodeType::kTerminalNode ||

892  ProductionNodeType::kOperatorNode);

894  // 使用ActionAndAttachedDataPointerEqualTo来判断两个转移结果是否相同
895  // 键值为ShiftAttachedData移入符号后转移到的语法分析表条目ID
896  std::unordered_map<SyntaxAnalysisTableEntryId,
897  std::list<SyntaxAnalysisTableEntryId>>
898  shift_classify_table;
899  // 键值为ReductAttachedData规约时使用的包装Reduct函数的类的对象的ID
900  // 这个ID是唯一的规约数据，所以可以用来标识规约数据
901  std::unordered_map<ProcessFunctionClassId,
902  std::list<SyntaxAnalysisTableEntryId>>
903  reduct_classify_table;
904  // 键值前半部分为ShiftReductAttachedData移入符号后转移到的语法分析表条目ID
905  // 键值后半部分为ReductAttachedData规约时使用的包装Reduct函数的类的对象ID
906  std::unordered_map<
907  std::pair<SyntaxAnalysisTableEntryId, ProcessFunctionClassId>,
908  std::list<SyntaxAnalysisTableEntryId>,
909  SyntaxAnalysisTableEntryIdAndProcessFunctionClassIdHasher>
910  shift_reduct_classify_table;
911  // 在该条件下不能规约也不能移入的语法分析表条目ID
912  std::list<SyntaxAnalysisTableEntryId> nothing_to_do_entry_ids;
913  ProductionNodeId transform_id = terminal_node_ids[index];
914  for (auto syntax_analysis_table_entry_id : syntax_analysis_table_entry_ids) {
915  // 利用unordered_map进行分类
916  auto action_and_attahced_data =
917  GetSyntaxAnalysisTableEntry(syntax_analysis_table_entry_id)
918  .AtTerminalNode(transform_id);
919  if (action_and_attahced_data) [[unlikely]] {
920  // 在该向前看节点处存在
921  switch (action_and_attahced_data->GetActionType()) {
922  case ActionType::kShift:
923  shift_classify_table[action_and_attahced_data->GetShiftAttachedData()
924  .GetNextSyntaxAnalysisTableEntryId()]
925  .push_back(syntax_analysis_table_entry_id);
926  break;
927  case ActionType::kReduct:
928  reduct_classify_table[action_and_attahced_data
929  ->GetReductAttachedData()
930  .GetProcessFunctionClassId()]
931  .push_back(syntax_analysis_table_entry_id);
932  break;
933  case ActionType::kShiftReduct:
934  shift_reduct_classify_table
935  [std::make_pair(action_and_attahced_data->GetShiftAttachedData()
936  .GetNextSyntaxAnalysisTableEntryId(),
937  action_and_attahced_data->GetReductAttachedData()
938  .GetProcessFunctionClassId())]
939  .push_back(syntax_analysis_table_entry_id);
940  break;
941  default:
942  assert(false);
943  break;
944  }
945  } else {
946  nothing_to_do_entry_ids.push_back(syntax_analysis_table_entry_id);
947  }
948  }
949 
950  size_t next_index = index + 1;
951  // 处理在当前条件下移入的节点
952  for (auto& syntax_analysis_table_entry_ids : shift_classify_table) {
