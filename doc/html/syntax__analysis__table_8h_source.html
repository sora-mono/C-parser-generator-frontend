593 template <class AttachedData>
594 requires std::is_same_v<std::decay_t<AttachedData>,
595  SyntaxAnalysisTableEntry::ShiftAttachedData> ||
596  std::is_same_v<std::decay_t<AttachedData>,
597  SyntaxAnalysisTableEntry::ReductAttachedData> ||
598  std::is_same_v<std::decay_t<AttachedData>,
599  SyntaxAnalysisTableEntry::AcceptAttachedData>
600 void SyntaxAnalysisTableEntry::SetTerminalNodeActionAndAttachedData(
601  ProductionNodeId node_id, AttachedData&& attached_data) {
602  if constexpr (std::is_same_v<std::decay_t<AttachedData>,
603  SyntaxAnalysisTableEntry::AcceptAttachedData>) {
604  // 传播向前看符号后，根语法分析表条目移入内置根节点后到达的条目在向前看符号
605  // 是文件尾节点时执行规约操作，需要设置为接受操作
606  assert(action_and_attached_data_.find(node_id)->second->GetActionType() ==
607  ActionType::kReduct);
608  action_and_attached_data_[node_id] = std::make_unique<AcceptAttachedData>(
609  std::forward<AttachedData>(attached_data));
610  return;
611  } else {
612  static_assert(
613  !std::is_same_v<std::decay_t<AttachedData>, ShiftReductAttachedData>,
614  "该类型仅允许通过在已有一种动作的基础上补全缺少的另一半后转换得到，不允"
615  "许直接传入");
616  // 使用二义性文法，语法分析表某些情况下需要对同一个节点支持移入和规约操作并存
617  auto iter = action_and_attached_data_.find(node_id);
618  if (iter == action_and_attached_data_.end()) {
619  // 新插入转移节点
620  action_and_attached_data_.emplace(
621  node_id, std::make_unique<std::decay_t<AttachedData>>(
622  std::forward<AttachedData>(attached_data)));
623  } else {
624  // 待插入的转移条件已存在
625  // 获取已经存储的数据
626  ActionAndAttachedDataInterface& data_already_in = *iter->second;
627  // 要么修改已有的规约后得到的节点，要么补全移入/规约的另一部分（规约/移入）
628  // 不应修改已有的移入后转移到的语法分析表条目
629  switch (data_already_in.GetActionType()) {
630  case ActionType::kShift:
631  if constexpr (std::is_same_v<std::decay_t<AttachedData>,
632  ReductAttachedData>) {
633  // 提供的数据是规约数据，转换为ShiftReductAttachedData
634  iter->second = std::make_unique<ShiftReductAttachedData>(
635  std::move(iter->second->GetShiftAttachedData()),
636  std::forward<AttachedData>(attached_data));
637  } else {
638  // 提供的是移入数据，要求必须与已有的数据相同
639  static_assert(
640  std::is_same_v<std::decay_t<AttachedData>, ShiftAttachedData>);
641  // 检查提供的移入数据是否与已有的移入数据相同
642  // 不能写反，否则当data_already_in为ShiftReductAttachedData时
643  // 无法获得与其它附属数据比较的正确语义
644  if (!data_already_in.IsSameOrPart(attached_data)) [[unlikely]] {
645  // 设置相同条件不同产生式下规约
646  std::cerr << std::format(
647  "一个项集在相同条件下只能规约一种产生式，请参考该项"
648  "集求闭包时的输出信息检查产生式");
649  exit(-1);
650  }
651  }
652  break;
653  case ActionType::kReduct:
654  if constexpr (std::is_same_v<std::decay_t<AttachedData>,
655  ShiftAttachedData>) {
656  // 提供的数据为移入部分数据，转换为ShiftReductAttachedData
657  iter->second = std::make_unique<ShiftReductAttachedData>(
658  std::forward<AttachedData>(attached_data),
659  std::move(iter->second->GetReductAttachedData()));
660  } else {
661  // 提供的是移入数据，需要检查是否与已有的数据相同
662  static_assert(
663  std::is_same_v<std::decay_t<AttachedData>, ReductAttachedData>);
664  // 不能写反，否则当data_already_in为ShiftReductAttachedData时
665  // 无法获得与其它附属数据比较的正确语义
666  if (!data_already_in.IsSameOrPart(attached_data)) [[unlikely]] {
667  // 设置相同条件不同产生式下规约
668  std::cerr << std::format(
669  "一个项集在相同条件下只能规约一种产生式，请参考该项"
670  "集求闭包时的输出信息检查产生式");
671  exit(-1);
672  }
673  }
674  break;
675  case ActionType::kShiftReduct: {
676  if (!data_already_in.IsSameOrPart(attached_data)) [[unlikely]] {
677  // 设置相同条件不同产生式下规约
678  std::cerr << std::format(
679  "一个项集在相同条件下只能规约一种产生式，请参考该项"
680  "集求闭包时的输出信息检查产生式");
681  exit(-1);
682  }
683  break;
684  }
685  default:
686  assert(false);
687  break;
688  }
689  }
690  }
