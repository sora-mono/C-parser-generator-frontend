295 size_t StructType::GetTypeStoreSize() const {
296  // 扫描结构体以获取最大的成员
297  const auto& struct_members = GetStructureMembers();
298  // C语言空结构体大小为0
299  // C++空结构体大小为1
300  size_t biggest_member_size = 0;
301  for (const auto& member : struct_members) {
302  size_t member_size = member.first->GetTypeStoreSize();
303  if (member_size > biggest_member_size) {
304  biggest_member_size = member_size;
305  }
306  }
307  // 获取系统基本字节单位与结构体最大成员的最小值作为字节对齐的单位
308  size_t aligen_size = std::min(size_t(4), biggest_member_size);
309  // C语言空结构体大小为0
310  size_t struct_size = 0;
311  // 填写结构体大小
312  for (const auto& member : struct_members) {
313  // 距离对齐剩余的空间
314  size_t rest_space_to_member_align = struct_size % aligen_size;
315  // 存储该成员需要的空间
316  size_t type_store_size = member.first->TypeSizeOf();
317  // 如果该成员所需空间大于距离对齐的空间，则这部分空间作为填充空间
318  if (rest_space_to_member_align < type_store_size) {
319  struct_size += aligen_size - rest_space_to_member_align;
320  }
321  // 已经对齐，增加存储所需的空间
322  struct_size += type_store_size;
323  }
324  return struct_size;

580  case StructOrBasicType::kPointer:
581  // 指针与指针组合
582  return std::make_pair(
583  std::shared_ptr<const TypeInterface>(),
584  DeclineMathematicalComputeTypeResult::kLeftRightBothPointer);
585  break;
586  default:
587  return std::make_pair(
588  std::shared_ptr<const TypeInterface>(),
589  DeclineMathematicalComputeTypeResult::kRightNotComputableType);
590  break;
591  }
592  } break;
593  default:
594  return std::make_pair(
595  std::shared_ptr<const TypeInterface>(),
596  DeclineMathematicalComputeTypeResult::kLeftNotComputableType);
597  break;
598  }

601 AddTypeResult TypeSystem::TypeData::AddType(
602  const std::shared_ptr<const TypeInterface>& type_to_add) {
603  std::monostate* empty_status_pointer =
604  std::get_if<std::monostate>(&type_data_);
605  if (empty_status_pointer != nullptr) [[likely]] {
606  // 该节点未保存任何指针
607  // 存入第一个指针
608  type_data_ = type_to_add;
609  return AddTypeResult::kNew;
610  } else {
611  std::shared_ptr<const TypeInterface>* shared_pointer =
612  std::get_if<std::shared_ptr<const TypeInterface>>(&type_data_);
613  if (shared_pointer != nullptr) [[likely]] {
614  // 该节点保存了一个指针，新增一个后转为vector存储
615  // 检查要添加的类型是否与已有类型重复或者重复添加kPointer/kBasic一类
616  if (IsSameKind((*shared_pointer)->GetType(), type_to_add->GetType()))
617  [[unlikely]] {
618  // 检查是否在添加函数类型
619  if (type_to_add->GetType() == StructOrBasicType::kFunction)
620  [[unlikely]] {
621  AddTypeResult function_define_add_result =
622  CheckFunctionDefineAddResult(
623  static_cast<const FunctionType&>(**shared_pointer),
624  static_cast<const FunctionType&>(*type_to_add));
625  if (function_define_add_result == AddTypeResult::kTypeAlreadyIn)
626  [[likely]] {
627  // 如果函数签名不同应返回AddTypeResult::kOverrideFunction
628  // 使用新的指针替换原有指针，从而更新函数参数名
629  *shared_pointer = type_to_add;
630  function_define_add_result = AddTypeResult::kFunctionDefine;
631  }
632  return function_define_add_result;
633  } else {
634  // 待添加的类型与已有类型相同，产生冲突
635  return AddTypeResult::kTypeAlreadyIn;
636  }
637  }
638  auto pointers =
639  std::make_unique<std::list<std::shared_ptr<const TypeInterface>>>();
640  // 将StructOrBasicType::kBasic或StructOrBasicType::kPointer类型放在最前面
641  // 从而优化查找类型的速度
642  if (IsSameKind(type_to_add->GetType(), StructOrBasicType::kBasic)) {
643  pointers->emplace_back(type_to_add);
644  pointers->emplace_back(std::move(*shared_pointer));
645  } else {
646  // 原来的类型不一定属于该大类，但新添加的类型一定不属于该大类
647  pointers->emplace_back(std::move(*shared_pointer));
648  pointers->emplace_back(type_to_add);
649  }
650  type_data_ = std::move(pointers);
651  return AddTypeResult::kShiftToVector;
652  } else {
653  // 该节点已经使用list存储
654  std::unique_ptr<std::list<std::shared_ptr<const TypeInterface>>>&
655  pointers = *std::get_if<
656  std::unique_ptr<std::list<std::shared_ptr<const TypeInterface>>>>(
657  &type_data_);
658  assert(pointers != nullptr);
659  // 检查待添加的类型与已存在的类型是否属于同一大类
660  for (auto& stored_pointer : *pointers) {
661  if (IsSameKind(stored_pointer->GetType(), type_to_add->GetType()))
662  [[unlikely]] {
663  // 检查是否在添加函数类型
664  if (type_to_add->GetType() == StructOrBasicType::kFunction)
665  [[unlikely]] {
666  AddTypeResult function_define_add_result =
667  CheckFunctionDefineAddResult(
668  static_cast<const FunctionType&>(*stored_pointer),
669  static_cast<const FunctionType&>(*type_to_add));
670  if (function_define_add_result == AddTypeResult::kTypeAlreadyIn)
671  [[likely]] {
672  // 这种情况为在函数定义前已添加函数声明
673  // 使用新的指针替换原有指针
674  stored_pointer = type_to_add;
675  return AddTypeResult::kFunctionDefine;
676  }
677  return function_define_add_result;
678  }
679  // 待添加的类型与已有类型相同，产生冲突
680  return AddTypeResult::kTypeAlreadyIn;
681  }
682  }
683  pointers->emplace_back(type_to_add);
684  if (IsSameKind(type_to_add->GetType(), StructOrBasicType::kBasic)) {
685  // 新插入的类型为StructOrBasicType::kBasic/kPointer
686  // 将该类型放到第一个位置以便优化无类型倾向性时的查找逻辑
687  std::swap(pointers->front(), pointers->back());
688  }
689  return AddTypeResult::kAddToVector;
690  }
691  }

694 std::pair<std::shared_ptr<const TypeInterface>, GetTypeResult>
695 TypeSystem::TypeData::GetType(StructOrBasicType type_prefer) const {
696  if (std::get_if<std::monostate>(&type_data_) != nullptr) [[unlikely]] {
697  // 未存储任何节点，等效于该名称对应的类型不存在
698  return std::make_pair(std::shared_ptr<TypeInterface>(),
699  GetTypeResult::kTypeNameNotFound);
700  }
701  const std::shared_ptr<const TypeInterface>* shared_pointer =
702  std::get_if<std::shared_ptr<const TypeInterface>>(&type_data_);
703  if (shared_pointer != nullptr) [[likely]] {
704  // 仅存储1个指针
705  if (type_prefer == StructOrBasicType::kNotSpecified ||
706  (*shared_pointer)->GetType() == type_prefer) [[likely]] {
707  // 无类型选择倾向或者类型选择倾向符合
708  return std::make_pair(*shared_pointer, GetTypeResult::kSuccess);
709  } else {
710  // 没有匹配的类型选择倾向
711  return std::make_pair(std::shared_ptr<TypeInterface>(),
712  GetTypeResult::kNoMatchTypePrefer);
713  }
714  } else {
715  // 存储多个指针
716  auto& pointers = *std::get_if<
717  std::unique_ptr<std::list<std::shared_ptr<const TypeInterface>>>>(
718  &type_data_);
719  // 判断是否指定类型选择倾向
720  if (type_prefer == StructOrBasicType::kNotSpecified) [[likely]] {
721  if (pointers->front()->GetType() == StructOrBasicType::kBasic) {
722  // 优先匹配kBasic
723  return std::make_pair(pointers->front(), GetTypeResult::kSuccess);
724  } else {
725  // vector中没有kBasic，并且一定存储至少两个指针，这些指针同级
726  return std::make_pair(std::shared_ptr<const TypeInterface>(),
727  GetTypeResult::kSeveralSameLevelMatches);
728  }
729  } else {
730  // 遍历vector搜索与给定类型选择倾向相同的指针
731  for (auto& pointer : *pointers) {
732  if (pointer->GetType() == type_prefer) [[unlikely]] {
733  return std::make_pair(pointer, GetTypeResult::kSuccess);
734  }
735  }
736  // 没有匹配的类型选择倾向
737  return std::make_pair(std::shared_ptr<const TypeInterface>(),
738  GetTypeResult::kNoMatchTypePrefer);
739  }
740  }

743 bool TypeSystem::TypeData::IsSameKind(StructOrBasicType type1,
744  StructOrBasicType type2) {
745  unsigned long long type1_ = static_cast<unsigned long long>(type1);
746  unsigned long long type2_ = static_cast<unsigned long long>(type2);
747  constexpr unsigned long long kBasicType =
748  static_cast<unsigned long long>(StructOrBasicType::kBasic);
749  constexpr unsigned long long kPointerType =
750  static_cast<unsigned long long>(StructOrBasicType::kPointer);
751  // 第二部分的运算用来检测是否均属于kPointer/kBasic
752  return type1 == type2 || ((type1_ ^ kBasicType) |
753  (type1_ ^ kPointerType) & (type2_ ^ kBasicType) |
754  (type2_ ^ kPointerType));
755 }

767 bool StructureTypeInterface::IsSameObject(
768  const TypeInterface& type_interface) const {
769  // this == &type_interface是优化手段
770  // 类型系统设计思路是尽可能多的共享一条类型链

785  return structure_name_this == structure_name_another;
786  }
787  }
788  // 二者都是匿名结构
789  // 如果匿名结构相同则二者使用相同的类型指针，在第一个if就返回
790  }
791  return false;
